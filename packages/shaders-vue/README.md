# @paper-design/shaders-react

## Usage

```
import { MeshGradient } from '@paper-design/shaders-react';

<MeshGradient
  colors={['#283BFC', '#FF2828', '#ddd']}
  blur="0.6"
  frequency="0.8"
  animationSpeed="1"
  style={{ width: 500, height: 200 }} />

// these settings can be configured in code or designed in Paper
```
